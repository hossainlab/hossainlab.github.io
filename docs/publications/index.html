// Publication filter functionality
{
  const filterButtons = document.querySelectorAll('.filter-btn');
  const publicationItems = document.querySelectorAll('.publication-item');

  // Filter function
  function filterPublications(filterValue) {
    publicationItems.forEach(item => {
      if (filterValue === 'all' || item.classList.contains(filterValue)) {
        item.classList.remove('hidden');
        item.classList.add('filtering-in');
        item.classList.remove('filtering-out');
      } else {
        item.classList.add('filtering-out');
        item.classList.remove('filtering-in');
        setTimeout(() => {
          item.classList.add('hidden');
        }, 300);
      }
    });

    // Update active button
    filterButtons.forEach(btn => {
      btn.classList.remove('active');
    });
    document.querySelector(`[data-filter="${filterValue}"]`).classList.add('active');

    // Show result count
    updateResultCount(filterValue);
  }

  // Result counter
  function updateResultCount(filterValue) {
    const visibleItems = document.querySelectorAll(`.publication-item${filterValue === 'all' ? '' : '.' + filterValue}`);
    const resultContainer = document.querySelector('.filter-results');

    if (!resultContainer) {
      const container = document.createElement('div');
      container.className = 'filter-results';
      document.querySelector('.publications-container').parentNode.insertBefore(container, document.querySelector('.publications-container'));
    }

    const count = visibleItems.length;
    const resultText = count === 1 ? '1 publication found' : `${count} publications found`;
    document.querySelector('.filter-results').textContent = filterValue === 'all' ? '' : resultText;
  }

  // Add event listeners
  filterButtons.forEach(button => {
    button.addEventListener('click', () => {
      const filterValue = button.getAttribute('data-filter');
      filterPublications(filterValue);
    });
  });

  return "Publication filters initialized";
}
